## 三、意图向量获取方案

针对你的对抗社区搜索场景，我设计了**三种方案**，从简单到复杂：

### 方案一：基于预训练模型的意图编码（推荐入门）

**核心思想**：直接使用预训练的文本编码器将查询转换为向量

```python
from sentence_transformers import SentenceTransformer
import torch
import torch.nn as nn

class PretrainedIntentEncoder(nn.Module):
    """
    基于预训练模型的意图编码器
    使用Sentence-BERT或BGE等模型
    """
    def __init__(self, model_name='BAAI/bge-large-zh-v1.5', intent_dim=256):
        super().__init__()
        
        # 预训练文本编码器
        self.text_encoder = SentenceTransformer(model_name)
        self.text_dim = self.text_encoder.get_sentence_embedding_dimension()
        
        # 意图投影层（将文本向量投影到意图空间）
        self.intent_projector = nn.Sequential(
            nn.Linear(self.text_dim, self.text_dim // 2),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(self.text_dim // 2, intent_dim),
            nn.LayerNorm(intent_dim)
        )
        
    def forward(self, query_text):
        """
        Args:
            query_text: 用户查询文本（字符串或列表）
        Returns:
            intent_vector: 意图向量 [batch_size, intent_dim]
        """
        # 编码文本
        with torch.no_grad():
            text_emb = self.text_encoder.encode(
                query_text, 
                convert_to_tensor=True,
                normalize_embeddings=True
            )
        
        # 投影到意图空间
        intent_vector = self.intent_projector(text_emb)
        
        return intent_vector


# 使用示例
encoder = PretrainedIntentEncoder()
query = "找出在社交媒体上通过隐蔽方式传播虚假信息的群体"
intent_vector = encoder(query)
print(f"意图向量维度: {intent_vector.shape}")  # [1, 256]
```

**优点**：简单、快速、无需训练
**缺点**：无法融入领域知识

### 方案二：对抗知识增强的意图编码（推荐采用）

**核心思想**：融合用户查询和对抗行为知识库，生成增强型意图向量

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from sentence_transformers import SentenceTransformer

class AdversarialKnowledgeEnhancedIntentEncoder(nn.Module):
    """
    对抗知识增强的意图编码器
    
    融合三种信息源：
    1. 用户查询语义
    2. 对抗行为模式库
    3. 历史案例特征
    """
    def __init__(self, 
                 model_name='BAAI/bge-large-zh-v1.5',
                 intent_dim=256,
                 num_adv_patterns=50,
                 pattern_dim=128):
        super().__init__()
        
        # ========== 组件1: 查询编码器 ==========
        self.query_encoder = SentenceTransformer(model_name)
        self.query_dim = self.query_encoder.get_sentence_embedding_dimension()
        
        # ========== 组件2: 对抗模式嵌入 ==========
        # 预定义的对抗行为模式嵌入表
        self.adv_pattern_embeddings = nn.Embedding(num_adv_patterns, pattern_dim)
        
        # 对抗模式描述（用于匹配）
        self.adv_pattern_descriptions = [
            "短时间内集中转发特定话题",
            "连接稀疏但语义高度相似",
            "使用隐语暗号进行沟通",
            "协同账号活动时间高度一致",
            "刻意删除历史互动记录",
            "伪装成普通用户的机器人账号",
            "通过间接路径传递敏感信息",
            "群组快速建立后迅速解散",
            "使用多个小号相互关注",
            "发布内容与个人信息不符",
            # ... 更多模式
        ]
        
        # ========== 组件3: 知识融合模块 ==========
        # 查询-模式注意力
        self.query_pattern_attention = nn.MultiheadAttention(
            embed_dim=pattern_dim,
            num_heads=4,
            dropout=0.1,
            batch_first=True
        )
        
        # 查询投影（对齐维度）
        self.query_projector = nn.Linear(self.query_dim, pattern_dim)
        
        # ========== 组件4: 意图融合器 ==========
        self.intent_fusion = nn.Sequential(
            nn.Linear(pattern_dim * 2, pattern_dim),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(pattern_dim, intent_dim),
            nn.LayerNorm(intent_dim)
        )
        
        # ========== 组件5: 置信度估计器 ==========
        self.confidence_estimator = nn.Sequential(
            nn.Linear(pattern_dim * 2, pattern_dim // 2),
            nn.ReLU(),
            nn.Linear(pattern_dim // 2, 1),
            nn.Sigmoid()
        )
        
        # 预编码对抗模式描述
        self._precompute_pattern_embeddings()
        
    def _precompute_pattern_embeddings(self):
        """预计算对抗模式的文本嵌入"""
        with torch.no_grad():
            self.pattern_text_embeddings = self.query_encoder.encode(
                self.adv_pattern_descriptions,
                convert_to_tensor=True,
                normalize_embeddings=True
            )
    
    def match_adv_patterns(self, query_emb, top_k=5):
        """
        匹配与查询最相关的对抗模式
        
        Args:
            query_emb: 查询嵌入 [batch_size, query_dim]
            top_k: 返回最相关的k个模式
        
        Returns:
            matched_indices: 匹配的模式索引
            match_scores: 匹配分数
        """
        # 计算与所有模式的相似度
        query_norm = F.normalize(query_emb, dim=-1)
        pattern_norm = F.normalize(self.pattern_text_embeddings, dim=-1)
        
        similarities = torch.mm(query_norm, pattern_norm.t())  # [batch, num_patterns]
        
        # 选择top-k
        match_scores, matched_indices = torch.topk(similarities, k=top_k, dim=-1)
        
        return matched_indices, match_scores
    
    def forward(self, query_text, return_details=False):
        """
        生成增强型意图向量
        
        Args:
            query_text: 用户查询文本
            return_details: 是否返回中间结果
        
        Returns:
            intent_vector: 增强型意图向量 [batch_size, intent_dim]
            (optional) details: 包含置信度、匹配模式等
        """
        # Step 1: 编码用户查询
        with torch.no_grad():
            query_emb = self.query_encoder.encode(
                query_text if isinstance(query_text, list) else [query_text],
                convert_to_tensor=True,
                normalize_embeddings=True
            )
        
        batch_size = query_emb.shape[0]
        device = query_emb.device
        
        # Step 2: 匹配相关对抗模式
        matched_indices, match_scores = self.match_adv_patterns(query_emb, top_k=5)
        
        # Step 3: 获取匹配模式的嵌入
        # 将pattern embeddings移到正确的设备
        pattern_embs = self.adv_pattern_embeddings(matched_indices.to(device))  # [batch, top_k, pattern_dim]
        
        # Step 4: 查询-模式注意力融合
        query_proj = self.query_projector(query_emb).unsqueeze(1)  # [batch, 1, pattern_dim]
        
        # 使用注意力机制融合模式信息
        attended_patterns, attention_weights = self.query_pattern_attention(
            query_proj,  # Query
            pattern_embs,  # Key
            pattern_embs   # Value
        )
        attended_patterns = attended_patterns.squeeze(1)  # [batch, pattern_dim]
        
        # Step 5: 融合生成增强型意图
        query_proj_squeezed = query_proj.squeeze(1)
        concat_features = torch.cat([query_proj_squeezed, attended_patterns], dim=-1)
        
        intent_vector = self.intent_fusion(concat_features)
        
        # Step 6: 估计置信度
        confidence = self.confidence_estimator(concat_features)
        
        if return_details:
            return intent_vector, {
                'confidence': confidence,
                'matched_patterns': matched_indices,
                'match_scores': match_scores,
                'attention_weights': attention_weights,
                'query_embedding': query_emb
            }
        
        return intent_vector


# 使用示例
encoder = AdversarialKnowledgeEnhancedIntentEncoder()
query = "找出在社交媒体上通过隐蔽方式协同传播虚假信息的群体"

intent_vector, details = encoder(query, return_details=True)
print(f"意图向量维度: {intent_vector.shape}")
print(f"置信度: {details['confidence'].item():.4f}")
print(f"匹配的对抗模式索引: {details['matched_patterns']}")
```

**优点**：融入领域知识，针对性强
**缺点**：需要构建对抗模式库

### 方案三：LLM增强的意图理解（最强方案）

**核心思想**：利用大语言模型的强大理解能力，结合结构化提示提取深层意图

```python
import torch
import torch.nn as nn
from transformers import AutoTokenizer, AutoModel
import json

class LLMEnhancedIntentEncoder(nn.Module):
    """
    LLM增强的意图编码器
    
    使用大语言模型进行：
    1. 意图分解：将复杂查询分解为多个子意图
    2. 对抗行为推理：推断可能的对抗行为模式
    3. 结构化意图提取：生成结构化的意图表示
    """
    def __init__(self, 
                 llm_model_name='Qwen/Qwen2-7B-Instruct',  # 或其他中文LLM
                 encoder_model_name='BAAI/bge-large-zh-v1.5',
                 intent_dim=256,
                 use_api=True,  # 是否使用API（如OpenAI/Claude）
                 api_key=None):
        super().__init__()
        
        self.use_api = use_api
        self.intent_dim = intent_dim
        
        # 文本编码器（用于将结构化输出编码为向量）
        self.text_encoder = AutoModel.from_pretrained(encoder_model_name)
        self.tokenizer = AutoTokenizer.from_pretrained(encoder_model_name)
        
        # 如果使用本地LLM
        if not use_api:
            self.llm_tokenizer = AutoTokenizer.from_pretrained(llm_model_name)
            self.llm_model = AutoModel.from_pretrained(llm_model_name)
        else:
            self.api_key = api_key
            
        # 意图向量投影
        self.intent_projector = nn.Sequential(
            nn.Linear(1024, 512),  # BGE-large输出维度
            nn.ReLU(),
            nn.Linear(512, intent_dim),
            nn.LayerNorm(intent_dim)
        )
        
        # 多意图聚合
        self.intent_aggregator = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model=intent_dim, nhead=4, batch_first=True),
            num_layers=2
        )
        
    def _build_intent_extraction_prompt(self, query):
        """
        构建意图提取的结构化提示
        """
        prompt = f"""你是一个网络安全分析专家，需要分析用户的搜索意图，以便在社交网络中发现隐蔽的对抗性群体。

用户查询："{query}"

请按以下JSON格式输出分析结果：

{{
    "primary_intent": "主要搜索意图（一句话描述）",
    "sub_intents": [
        "子意图1",
        "子意图2",
        "子意图3"
    ],
    "target_behavior_patterns": [
        {{
            "pattern_name": "行为模式名称",
            "description": "具体描述",
            "detection_signals": ["信号1", "信号2"]
        }}
    ],
    "adversarial_characteristics": [
        "对抗特征1：描述",
        "对抗特征2：描述"
    ],
    "structural_indicators": [
        "结构指标1：如连接稀疏性",
        "结构指标2：如时序异常"
    ],
    "intent_keywords": ["关键词1", "关键词2", "关键词3"],
    "risk_level": "high/medium/low",
    "confidence": 0.85
}}

只输出JSON，不要有其他内容。"""
        
        return prompt
    
    def _call_llm_api(self, prompt):
        """
        调用LLM API（示例使用OpenAI格式）
        实际使用时替换为你使用的API
        """
        # 这里是伪代码，需要替换为实际的API调用
        # 例如使用 openai, anthropic, 或本地部署的模型
        
        # 示例：使用requests调用本地Ollama
        import requests
        
        response = requests.post(
            'http://localhost:11434/api/generate',
            json={
                'model': 'qwen2:7b',
                'prompt': prompt,
                'stream': False
            }
        )
        
        return response.json()['response']
    
    def _parse_llm_output(self, llm_output):
        """
        解析LLM输出的JSON
        """
        try:
            # 尝试直接解析
            intent_data = json.loads(llm_output)
        except json.JSONDecodeError:
            # 尝试提取JSON部分
            import re
            json_match = re.search(r'\{.*\}', llm_output, re.DOTALL)
            if json_match:
                intent_data = json.loads(json_match.group())
            else:
                # 返回默认结构
                intent_data = {
                    "primary_intent": llm_output[:100],
                    "sub_intents": [],
                    "target_behavior_patterns": [],
                    "adversarial_characteristics": [],
                    "structural_indicators": [],
                    "intent_keywords": [],
                    "risk_level": "medium",
                    "confidence": 0.5
                }
        
        return intent_data
    
    def _encode_intent_components(self, intent_data):
        """
        将意图的各个组件编码为向量
        """
        components_to_encode = []
        
        # 主意图
        components_to_encode.append(intent_data.get('primary_intent', ''))
        
        # 子意图
        components_to_encode.extend(intent_data.get('sub_intents', []))
        
        # 对抗特征
        components_to_encode.extend(intent_data.get('adversarial_characteristics', []))
        
        # 结构指标
        components_to_encode.extend(intent_data.get('structural_indicators', []))
        
        # 行为模式描述
        for pattern in intent_data.get('target_behavior_patterns', []):
            components_to_encode.append(pattern.get('description', ''))
        
        # 过滤空字符串
        components_to_encode = [c for c in components_to_encode if c.strip()]
        
        if not components_to_encode:
            components_to_encode = [intent_data.get('primary_intent', 'unknown intent')]
        
        # 使用文本编码器编码
        inputs = self.tokenizer(
            components_to_encode,
            padding=True,
            truncation=True,
            max_length=512,
            return_tensors='pt'
        )
        
        with torch.no_grad():
            outputs = self.text_encoder(**inputs)
            # 使用[CLS] token或mean pooling
            embeddings = outputs.last_hidden_state[:, 0, :]  # [num_components, hidden_dim]
        
        return embeddings, intent_data.get('confidence', 0.5)
    
    def forward(self, query_text, return_details=False):
        """
        生成LLM增强的意图向量
        """
        # Step 1: 构建提示并调用LLM
        prompt = self._build_intent_extraction_prompt(query_text)
        
        if self.use_api:
            llm_output = self._call_llm_api(prompt)
        else:
            # 使用本地模型
            inputs = self.llm_tokenizer(prompt, return_tensors='pt')
            outputs = self.llm_model.generate(**inputs, max_length=1024)
            llm_output = self.llm_tokenizer.decode(outputs[0])
        
        # Step 2: 解析LLM输出
        intent_data = self._parse_llm_output(llm_output)
        
        # Step 3: 编码意图组件
        component_embeddings, confidence = self._encode_intent_components(intent_data)
        
        # Step 4: 投影到意图空间
        intent_components = self.intent_projector(component_embeddings)  # [num_components, intent_dim]
        
        # Step 5: 聚合多个意图组件
        # 添加batch维度
        intent_components = intent_components.unsqueeze(0)  # [1, num_components, intent_dim]
        aggregated = self.intent_aggregator(intent_components)
        
        # 使用mean pooling获取最终意图向量
        intent_vector = aggregated.mean(dim=1)  # [1, intent_dim]
        
        if return_details:
            return intent_vector, {
                'intent_data': intent_data,
                'confidence': confidence,
                'num_components': component_embeddings.shape[0],
                'llm_output': llm_output
            }
        
        return intent_vector


# 使用示例（需要配置LLM API）
"""
encoder = LLMEnhancedIntentEncoder(use_api=True)
query = "找出在社交媒体上通过隐蔽方式协同传播历史虚无主义内容的群体"

print(f"意图向量维度: {intent_vector.shape}")
print(f"LLM识别的意图数据: {json.dumps(details['intent_data'], ensure_ascii=False, indent=2)}")
"""
```

**优点**：理解能力强，可处理复杂查询
**缺点**：计算成本高，需要LLM支持

## 四、推荐方案：混合意图编码器

结合上述三种方案的优点，设计**混合意图编码器**：

```python
class HybridIntentEncoder(nn.Module):
    """
    混合意图编码器
    
    结合：
    1. 预训练模型的快速编码能力
    2. 对抗知识库的领域知识
    3. LLM的深层理解能力（可选）
    """
    def __init__(self, 
                 intent_dim=256,
                 use_llm=False,
                 use_knowledge_base=True):
        super().__init__()
        
        self.intent_dim = intent_dim
        self.use_llm = use_llm
        
        # 基础编码器（始终使用）
        self.base_encoder = PretrainedIntentEncoder(intent_dim=intent_dim)
        
        # 知识增强编码器
        if use_knowledge_base:
            self.knowledge_encoder = AdversarialKnowledgeEnhancedIntentEncoder(
                intent_dim=intent_dim
            )
        else:
            self.knowledge_encoder = None
            
        # LLM编码器（可选）
        if use_llm:
            self.llm_encoder = LLMEnhancedIntentEncoder(intent_dim=intent_dim)
        else:
            self.llm_encoder = None
        
        # 多源意图融合
        num_sources = 1 + int(use_knowledge_base) + int(use_llm)
        self.fusion_layer = nn.Sequential(
            nn.Linear(intent_dim * num_sources, intent_dim * 2),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(intent_dim * 2, intent_dim),
            nn.LayerNorm(intent_dim)
        )
        
        # 置信度加权
        self.confidence_weights = nn.Linear(num_sources, num_sources)
        
    def forward(self, query_text, return_details=False):
        """
        生成混合意图向量
        """
        intent_vectors = []
        confidences = []
        details = {}
        
        # 1. 基础编码
        base_intent = self.base_encoder(query_text)
        intent_vectors.append(base_intent)
        confidences.append(torch.tensor([0.8]))  # 基础置信度
        details['base_intent'] = base_intent
        
        # 2. 知识增强编码
        if self.knowledge_encoder is not None:
            knowledge_intent, kb_details = self.knowledge_encoder(
                query_text, return_details=True
            )
            intent_vectors.append(knowledge_intent)
            confidences.append(kb_details['confidence'])
            details['knowledge_intent'] = knowledge_intent
            details['knowledge_details'] = kb_details
        
        # 3. LLM增强编码
        if self.llm_encoder is not None:
            llm_intent, llm_details = self.llm_encoder(
                query_text, return_details=True
            )
            intent_vectors.append(llm_intent)
            confidences.append(torch.tensor([llm_details['confidence']]))
            details['llm_intent'] = llm_intent
            details['llm_details'] = llm_details
        
        # 4. 融合多源意图
        # 拼接所有意图向量
        concat_intents = torch.cat(intent_vectors, dim=-1)
        
        # 生成最终意图向量
        final_intent = self.fusion_layer(concat_intents)
        
        # 计算综合置信度
        confidences_tensor = torch.cat(confidences).unsqueeze(0)
        weighted_confidence = torch.sigmoid(self.confidence_weights(confidences_tensor))
        overall_confidence = weighted_confidence.mean()
        
        details['final_intent'] = final_intent
        details['overall_confidence'] = overall_confidence
        
        if return_details:
            return final_intent, details
        
        return final_intent
```

## 五、对抗知识库构建

意图向量的质量很大程度上取决于对抗知识库的质量：

### 5.1 知识库结构

```python
ADVERSARIAL_KNOWLEDGE_BASE = {
    # ========== 行为模式 ==========
```

```python
def preprocess_knowledge_base(kb, encoder):
    """
    预处理知识库，计算所有文本的嵌入向量
    """
    # 编码行为模式
    for pattern in kb['behavior_patterns']:
        text = f"{pattern['name']}: {pattern['description']}"
        pattern['embedding'] = encoder.encode(text)
    
    # 编码结构特征
    for feature in kb['structural_features']:
        text = f"{feature['name']}: {feature['description']}"
        feature['embedding'] = encoder.encode(text)
    
    # 编码历史案例
    for case in kb['historical_cases']:
        text = f"{case['name']}: {case['description']}"
        case['embedding'] = encoder.encode(text)
    
    return kb
```

### 6.1 意图编码器的训练

如果有标注数据，可以使用对比学习训练意图编码器：

```python
class IntentContrastiveLoss(nn.Module):
    """
    意图对比学习损失
    
    正样本：同一意图的不同表述
    负样本：不同意图的表述
    """
    def __init__(self, temperature=0.07):
        super().__init__()
        self.temperature = temperature
        
    def forward(self, intent_vectors, labels):
        """
        Args:
            intent_vectors: [batch_size, intent_dim]
            labels: [batch_size] 意图类别标签
        """
        # 归一化
        intent_vectors = F.normalize(intent_vectors, dim=1)
        
        # 计算相似度矩阵
        sim_matrix = torch.mm(intent_vectors, intent_vectors.t()) / self.temperature
        
        # 构建标签矩阵
        labels = labels.unsqueeze(0)
        mask = torch.eq(labels, labels.t()).float()
        
        # 移除对角线
        mask = mask - torch.eye(mask.shape[0], device=mask.device)
        
        # 计算对比损失
        exp_sim = torch.exp(sim_matrix)
        log_prob = sim_matrix - torch.log(exp_sim.sum(dim=1, keepdim=True))
        
        # 只计算正样本的损失
        mean_log_prob_pos = (mask * log_prob).sum(dim=1) / (mask.sum(dim=1) + 1e-8)
        loss = -mean_log_prob_pos.mean()
        
        return loss
```

```python
def train_intent_encoder_unsupervised(encoder, queries, epochs=100):
    """
    无监督训练意图编码器
    使用数据增强生成正样本对
    """
    optimizer = torch.optim.Adam(encoder.parameters(), lr=1e-4)
    
    for epoch in range(epochs):
        for query in queries:

            # 编码
            intent1 = encoder(query)
            intent2 = encoder(augmented_query)
            
            # InfoNCE损失
            loss = infonce_loss(intent1, intent2)
            
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
```
